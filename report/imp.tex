		 $rhs$ the right hand side vector
	}
	\KwOut{Vector $x$ is the solution of $Ax = rhs$}
	{$v_1 = rhs$\;}

		\Proc{$B$-orthonormalize $V_k \rightarrow Z_k$}{
			Compute $\hat{B} = BV_k$\;
		}
	}
	\caption{Basic Trace Minimization}
\end{algorithm}
